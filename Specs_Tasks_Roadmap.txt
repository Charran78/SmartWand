    acceleration: tuple[float, float, float]
    gyroscope: tuple[float, float, float]

class SmartWandSimulator:
    """Simula el hardware del SmartWand para desarrollo"""

                    random.uniform(-3, 3)
                )

                data.acceleration = (8.5, 7.2, 3.8)
                await self._trigger_fall_detection()

from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from pydantic import BaseModel
from typing import List, Dict
import asyncio

app = FastAPI(title="SmartWand Backend")

class FallAlert(BaseModel):
    user_id: str
    timestamp: str
    location: Dict[str, float]  # GPS coordinates
    confidence: float
    sensor_data: Dict

class EmergencyContact(BaseModel):
    name: str
    phone: str
    relationship: str
    priority: int

@app.websocket("/ws/smartwand/{device_id}")
async def smartwand_websocket(websocket: WebSocket, device_id: str):
    """WebSocket para comunicaci√≥n en tiempo real con el SmartWand"""
    await websocket.accept()
    try:
        while True:
            data = await websocket.receive_json()
            
            # Procesar datos del wearable
            if data.get("type") == "sensor_data":
                await process_sensor_data(device_id, data)
            elif data.get("type") == "fall_detected":
                await trigger_emergency_protocol(device_id, data)
                
    except WebSocketDisconnect:
        print(f"Device {device_id} disconnected")

async def trigger_emergency_protocol(device_id: str, fall_data: Dict):
    """Protocolo de emergencia automatizado"""
    # 1. Notificar contactos de emergencia
    contacts = await get_emergency_contacts(device_id)
    for contact in contacts:
        await send_sms_alert(contact, device_id, fall_data)
    
    # 2. Registrar en base de datos
    await log_emergency_event(device_id, fall_data)
    
    # 3. Opcional: Notificar servicios de emergencia
    if fall_data.get("severity") == "high":
        await call_emergency_services(fall_data.get("location"))

// MainActivity.kt - Versi√≥n simplificada
class MainActivity : AppCompatActivity() {
    
    private lateinit var binding: ActivityMainBinding
    private val viewModel: SmartWandViewModel by viewModels()
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)
        
        setupObservers()
        setupControls()
    }
    
    private fun setupObservers() {
        viewModel.fallDetected.observe(this) { fallEvent ->
            if (fallEvent != null) {
                showEmergencyAlert(fallEvent)
            }
        }
        
        viewModel.iotDevices.observe(this) { devices ->
            updateDeviceControls(devices)
        }
    }
    
    private fun setupControls() {
        // Control de dispositivos IoT b√°sico
        binding.btnLightToggle.setOnClickListener {
            viewModel.toggleDevice("living_room_light")
        }
        
        binding.btnEmergencyCall.setOnClickListener {
            viewModel.initiateEmergencyCall()
        }
    }
}
